import matplotlib.pyplot as plt
from scipy.integrate import odeint
from numpy import*

def model(y,t0,r,K):
    # note t0 is not used, but required by odeint
    dY = y[0] * (r - y[0] / float(K))
    return [dY]     #note a list is returned
    

y = [0.1]   #initial value for N

#parameters, r and k
params = (0.1, 10) 

#parameters used by lg_2
p2 = (0.3, 50)

#paramaters used by lg_3
p3 = (0.7, 10)

log_growth = odeint(
    func=model,y0=y,
    t=range(0,100),args=params
    )
 
lg_2 = odeint(
    func=model,y0=3,
    t=range(0,100),args=p2
    )

lg_3 = odeint(
    func=model,y0=0.5,
    t=range(0,80),args=p3
    )

plt.plot(lg_2, 'g')
plt.plot(log_growth, 'b')
plt.plot(lg_3, 'r')
plt.xlabel('t')
plt.grid()
plt.show()

Based off the example given I experimented with several parameters of the odeint function provided by the scipy library. I noticed the parameter t changes the range on the x-axis of the graph. The function colored in red has a range of eighty which can be seen graphed above. Each growth model has different intitial starting values changing their starting point location along the y-axis.

import matplotlib.pyplot as plt
from scipy.integrate import odeint
import math

def model(y,time,r,K):
    N = y[0]
    dN = K * math.exp(r*time)
    return [dN]

y = [1]

params = (0.05, 10)

times = range(0,100)
out = odeint(func=model,y0=y,t=times,args=params)

out2 = odeint(func=model, y0=2, t=range(0,200), args=(0.02, 10))

out3 = odeint(func=model, y0=2, t=range(0,200), args=(0.03, -10))

plt.plot(out)
plt.plot(out2, 'r')
plt.plot(out3, 'g')
plt.show()

from pylab import *

def initialize():
    global x, y, xresult, yresult
    x = 0.6
    y = .09
    xresult = [x]
    yresult = [y]

def observe():
    global x, y, xresult, yresult
    xresult.append(x)
    yresult.append(y)

def update():
    global x, y, xresult, yresult
    nx = 0.7 * x + y
    ny = -0.3 * x + y
    #ny = 1 /(2)*y+1
    x,y = nx, ny

initialize()
for t in range(20):
    update()
    observe()

plot(xresult, 'b-')
plot(yresult, 'g--')

show()
